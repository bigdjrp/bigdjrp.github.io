#include <iostream>
#include <string>
#include <sstream> 
using namespace std;

class Ratio{
friend Ratio operator*(Ratio &x, Ratio &y);
friend Ratio operator+(Ratio &x, Ratio &y);
friend Ratio operator-(Ratio &x, Ratio &y);
friend Ratio operator/(Ratio &x, Ratio &y);
friend Ratio operator>(Ratio &x, Ratio &y);
friend Ratio operator<(Ratio &x, Ratio &y);
public:
Ratio(int n, int d);
int getNum();
int getDen();
void setNum(int n);
void setDen(int d);
void setgcd(int n);
double convert();
void invert();
void print();
int gcd(int m, int n);
int getgcd();
void reduce();
private:
int num;
int den;
int gc;
};


Ratio operator*(Ratio & x, Ratio & y){
Ratio z(x.num*y.num, x.den*y.den);
z.gcd(z.num, z.den);
z.reduce();
return z;
}

Ratio operator+(Ratio & x, Ratio & y){
Ratio z((x.num*y.den) + (y.num*x.den), x.den*y.den);
z.gcd(z.num, z.den);
z.reduce();
return z;
}

Ratio operator-(Ratio & x, Ratio & y){
Ratio z((x.num*y.den) - (y.num*x.den), x.den*y.den);
z.gcd(z.num, z.den);
z.reduce();
return z;
}

Ratio operator/(Ratio & x, Ratio & y){
Ratio z(x.num*y.den, x.den*y.num);
z.gcd(z.num, z.den);
z.reduce();
return z;
}

Ratio operator>(Ratio &x, Ratio &y){
double xx = (double)(x.getNum()/x.getDen());
double yy = (double)(y.getNum()/y.getDen());
if(xx > yy){
cout << "true" << endl;
}
else{
cout << "false" << endl;
}
}

Ratio operator<(Ratio &x, Ratio &y){
double xx = (double)(x.getNum()/x.getDen());
double yy = (double)(y.getNum()/y.getDen());
if(xx < yy){
cout << "true" << endl;
}
else{
cout << "false" << endl;
}
}

Ratio operator!=(Ratio &x, Ratio &y){
if (x.getNum() != y.getNum() || x.getDen() != y.getDen()){
cout << "true" << endl;
}
else{
cout << "false" << endl;
}
}

Ratio operator==(Ratio &x, Ratio &y){
if (x.getNum() == y.getNum() && x.getDen() == y.getDen()){
cout << "true" << endl;
}
else{
cout << "false" << endl;
}
}

Ratio::Ratio(int n, int d){ // ::->scope resolution operator
num = n;
den = d;
}

int Ratio::getNum(){
return num;
}

int Ratio::getDen(){
return den;
}

void Ratio::setNum(int n){
num = n;
}

void Ratio::setDen(int d){
den = d;
}

int Ratio::getgcd(){
return gc;
}

void Ratio::setgcd(int n){
gc = n;
}

void Ratio::reduce(){
num = (num/getgcd());
den = (den/getgcd());
}

int Ratio::gcd(int m, int n){
while(m != 0){
int temp = m; 
m = n%m; 
n = temp;
}
setgcd(n);
return n;
}

double Ratio::convert(){
return double(num)/den;
}

void Ratio::invert(){
int temp = num;
num = den;
den = temp;
}

void Ratio::print(){
cout << num << '/' << den;
}
